		$(function(){
			 var bgcolors = ["bgcolor1", "bgcolor2", "bgcolor3", "bgcolor4", "bgcolor5", "bgcolor6", "bgcolor7", "bgcolor8", "bgcolor9", "bgcolor10", "bgcolor11", "bgcolor12", "bgcolor13"];
			 var bbgcolors = ["bbgcolor3", "bbgcolor2", "bbgcolor1","bbgcolor2", "bbgcolor3", "bbgcolor1", "bbgcolor2", "bbgcolor3"];
			 var setNavBtn = function(){
			 	$(".article_box").each(function(){
		      		var $aBox = $(this),
		      			$wBoxList = $aBox.find(".widget_box_list"),
		      			$iBox = $aBox.find(".article_inner_box");
		      			if($iBox[0].scrollLeft>0){
		      				$aBox.find(".arrow_left").removeClass("hidden");
		      			}else{
		      				$aBox.find(".arrow_left").addClass("hidden");
		      			}
		      			if($iBox[0].scrollLeft < $wBoxList.width()-$iBox.width()){
		      				$aBox.find(".arrow_right").removeClass("hidden");
		      			}else{
		      				$aBox.find(".arrow_right").addClass("hidden");
		      			}
		      	});
			 }
	          //遍历随机颜色值
		      $(".widget_box .widget_list").each(function () {
		            var num = Math.floor(Math.random() * 13);
		            $(this).addClass(bgcolors[num]);
		      });
		      $(".article_box").each(function(i){
		            $(this).addClass(bbgcolors[i]);
		            $(this).find(".article_title").addClass(bbgcolors[i]).find("h3").css({"color":"white"});
		      });
		      //设置导航
		      $(".article_box").each(function(){
		      		var $abox = $(this);
		      		var $wboxs = $abox.find(".widget_box");
		      		var nw = $wboxs.length * $abox.width();
		      		$wboxs.parent().width(nw);
      				setNavBtn($abox);
		      });
		      setNavBtn();
		      $(".article_box .nav_arrow").click(function(){
		      		var $aBox = $(this).parent().parent(),
		      			$wBoxList = $aBox.find(".widget_box_list"),
		      			$iBox = $aBox.find(".article_inner_box");
		      		if($(this).hasClass("arrow_left")){//上一页，左移，显示右边
		      			var cur = $iBox[0].scrollLeft- 1000;
		      			if(cur<0){
		      				cur=0;
		      			}
		      			$iBox.animate({scrollLeft:cur+"px"},500,function(){
		      				setNavBtn();
		      			});
		      		}else{//右移，显示左边
		      			var cur = $iBox[0].scrollLeft + 1000;
		      			var mLen = $wBoxList.width()-$iBox.width();
		      			if(cur>mLen){
		      				cur=mLen;
		      			}
		      			$iBox.animate({scrollLeft:cur+"px"},500,function(){
		      				setNavBtn();
		      			});
		      			
		      		}
		      });
		});
	</script>
